    <script>
        let growthChartInstance = null;
        let history = [];

        // Format currency to INR format
        const formatCurrency = (amount) => {
            return new Intl.NumberFormat('en-IN', {
                style: 'currency',
                currency: 'INR',
                maximumFractionDigits: 0
            }).format(amount);
        };

        function calculateProjection() {
            // Get input values
            const principal = parseFloat(document.getElementById('principal').value);
            const rate = parseFloat(document.getElementById('rate').value);
            const years = parseInt(document.getElementById('years').value);

            if (isNaN(principal) || isNaN(rate) || isNaN(years)) return;

            // Math formula: A = P(1 + r/100)^t
            const futureValue = principal * Math.pow((1 + rate / 100), years);
            const totalInterest = futureValue - principal;

            // Update DOM Results
            document.getElementById('future-value').innerText = formatCurrency(futureValue);
            document.getElementById('total-interest').innerHTML = `
                <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="3">
                    <path stroke-linecap="round" stroke-linejoin="round" d="M13 7h8m0 0v8m0-8l-8 8-4-4-6 6" />
                </svg>
                ${formatCurrency(totalInterest)}
            `;
            document.getElementById('total-contributions').innerText = formatCurrency(principal);
            document.getElementById('chart-subtitle').innerText = `Visualizing your wealth accumulation over ${years} years at a ${rate}% annual rate.`;

            // Prepare Chart Data
            const labels = [];
            const dataPrincipal = [];
            const dataGrowth = [];

            for (let i = 0; i <= years; i++) {
                labels.push(`Yr ${i}`);
                dataPrincipal.push(principal);
                dataGrowth.push(principal * Math.pow((1 + rate / 100), i));
            }

            renderChart(labels, dataPrincipal, dataGrowth);
            addToHistory(principal, rate, years, futureValue);
        }

        function renderChart(labels, dataPrincipal, dataGrowth) {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            // Destroy existing chart if it exists
            if (growthChartInstance) {
                growthChartInstance.destroy();
            }

            growthChartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Total Value',
                            data: dataGrowth,
                            borderColor: '#10b981', // Emerald 500
                            backgroundColor: 'rgba(16, 185, 129, 0.1)',
                            borderWidth: 3,
                            fill: true,
                            tension: 0.4,
                            pointRadius: 0
                        },
                        {
                            label: 'Initial Investment',
                            data: dataPrincipal,
                            borderColor: '#64748b', // Slate 500
                            borderWidth: 2,
                            borderDash: [5, 5],
                            fill: false,
                            pointRadius: 0
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    return formatCurrency(context.raw);
                                }
                            }
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            grid: { color: '#f1f5f9', drawBorder: false },
                            ticks: {
                                callback: function(value) { return '₹' + (value / 1000) + 'k'; },
                                color: '#94a3b8'
                            }
                        },
                        x: {
                            grid: { display: false, drawBorder: false },
                            ticks: { color: '#94a3b8' }
                        }
                    }
                }
            });
        }

        function addToHistory(principal, rate, years, futureValue) {
            // Add to top of history array
            history.unshift({ principal, rate, years, futureValue });
            
            // Keep only last 3 items to match design
            if (history.length > 3) {
                history.pop();
            }

            renderHistory();
        }

        function renderHistory() {
            const container = document.getElementById('history-container');
            container.innerHTML = ''; // Clear current

            history.forEach(item => {
                const historyCard = document.createElement('div');
                historyCard.className = 'flex flex-col sm:flex-row sm:items-center justify-between p-4 bg-slate-50 border border-slate-100 rounded-xl';
                
                historyCard.innerHTML = `
                    <div class="mb-3 sm:mb-0">
                        <p class="text-xs font-bold text-slate-400 tracking-wider mb-2">INPUTS</p>
                        <div class="flex items-center gap-4 text-slate-700 text-sm font-medium">
                            <span class="flex items-center gap-1.5"><div class="w-1.5 h-1.5 rounded-full bg-emerald-300"></div> ${formatCurrency(item.principal)}</span>
                            <span class="flex items-center gap-1.5"><div class="w-1.5 h-1.5 rounded-full bg-emerald-300"></div> ${item.rate}% / yr</span>
                            <span class="flex items-center gap-1.5"><div class="w-1.5 h-1.5 rounded-full bg-emerald-300"></div> ${item.years} years</span>
                        </div>
                    </div>
                    <div class="text-left sm:text-right">
                        <p class="text-xs font-bold text-slate-400 tracking-wider mb-2">FUTURE VALUE</p>
                        <p class="text-lg font-bold text-emerald-500 flex items-center sm:justify-end gap-1">
                            <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="3">
                                <path stroke-linecap="round" stroke-linejoin="round" d="M13 7h8m0 0v8m0-8l-8 8-4-4-6 6" />
                            </svg>
                            ${formatCurrency(item.futureValue)}
                        </p>
                    </div>
                `;
                container.appendChild(historyCard);
            });
        }

        // Initialize on load
        window.onload = calculateProjection;

    </script>
